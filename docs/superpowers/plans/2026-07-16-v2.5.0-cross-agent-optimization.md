# ForgeWorks v2.5.0 Implementation Plan — cross-agent generalization, gate parity, token/limit economics

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Implement the approved spec at `docs/superpowers/specs/2026-07-16-v2.5.0-cross-agent-optimization-design.md`: Go `-race` in the gate, honest README positioning, a cross-profile parity contract in CI, an availability-aware B13 agent-selection question with a mutable `docs/agents.json` runtime roster + `/select-agents` skill, a haiku `@utility` subagent with session-limit workload shifting, and the v2.5.0 release mechanics.

**Architecture:** The template repo generates projects via a deterministic renderer (`init-project/render.py` + `render_schema.py`, stdlib Python, no template engine) from an interview answers file. Conditional emission is `skip_file()` rules; conditional text is snippet files under `templates/conditional/`; byte-level correctness is proven by golden fixtures (`.github/scripts/golden_test.py`). All changes follow those existing patterns.

**Tech Stack:** stdlib Python 3, bash, Markdown. No new dependencies.

## Global Constraints

- **Repo conventions (`AGENTS.md`):** Conventional Commits (`feat(template):` etc.), NO Co-Authored-By trailer, this work goes via a PR (it touches SKILL.md Phase 4 + core AGENTS.md blocks + bootstrap pins).
- **`templates/core/` stays language-free** — no toolchain names outside profiles/conditional blocks.
- **Renderer fails closed** — every new placeholder must be in SKILL.md's placeholder table AND `render.py`'s mapping; any leftover `{{...}}` in output is an error.
- **FW-BLOCK versions:** any edited rule block in `templates/core/AGENTS.md` bumps its opening marker version to `v2.5.0`.
- **ASCII only** in template files (match existing `--` style, no smart quotes/em dashes).
- **Determinism:** same answers file -> same bytes. No timestamps, no environment reads inside `render.py`.
- **Verification loop:** `python3 .github/scripts/golden_test.py` must pass at the END (Task 11 regenerates expected trees once); mid-branch tasks verify by rendering to a temp dir and inspecting.
- **Line caps:** the ~100/200-line cap applies to generated files (`templates/core/`, `templates/profiles/`), NOT to SKILL.md / golden_test.py.

```python
def skip_file(relpath: str, source: str, ans: dict) -> bool:
    parts = relpath.split(os.sep)
    if source == "profile" and relpath == "profile.json":
        return True  # renderer input, never part of a generated project
    if (parts[0] == ".claude" and not claude_selected(ans)):
        return True  # Claude Code not in the roster: no agents/hooks/skills/settings
    if parts[0] == ".devcontainer" and ans["stack"]["uses_devcontainer"] == "no":
        return True
    if parts[:2] == ["docs", "explanations"] and ans["opt_ins"]["explanations"] == "no":
        return True
    return relpath == os.path.join("docs", "memory.md") and ans["opt_ins"]["mem0"] == "no"
```

```python
    if claude_selected(ans):
        if os.path.lexists(claude_md):
            os.remove(claude_md)
        try:
            os.symlink("AGENTS.md", claude_md)
        except (OSError, NotImplementedError):

    for sub in (os.path.join(".claude", "hooks"), "scripts"):

    agents_json = os.path.join(out_dir, "docs", "agents.json")
```

```python
# Cross-profile parity contract: every profile must carry evidence of each
# quality capability. (relpath, needle) -- needle None means "file exists".
# race/sanitizer applies only where the language has one (Go: -race).
PARITY_CONTRACT = {
    "python": [
        ("scripts/qa.sh", "linecap.sh"),            # line cap
        ("scripts/qa.sh", "ruff format --check"),   # format check
        ("scripts/qa.sh", "ruff check"),            # lint as error
        ("scripts/qa.sh", "mypy"),                  # type check
        ("scripts/qa.sh", 'pytest -m "not e2e"'),   # unit gate
        ("scripts/e2e.sh", "pytest"),               # e2e runner
    ],
    "typescript": [
        ("eslint.config.js", "max-lines"),          # line cap (lint rule)
        ("package.json", "prettier --check"),       # format check
        ("package.json", "eslint ."),               # lint as error
        ("package.json", "tsc --noEmit"),           # type check
        ("package.json", "vitest run"),             # unit gate
        ("package.json", "playwright test"),        # e2e runner
    ],
    "go": [
        ("scripts/qa.sh", "linecap.sh"),
        ("scripts/qa.sh", "gofmt -l"),
        ("scripts/qa.sh", "golangci-lint run"),
        ("scripts/qa.sh", "go vet"),
        ("scripts/qa.sh", "go test -race ./..."),   # unit gate + race detector
        ("scripts/e2e.sh", "go test -race -tags e2e"),
    ],
    "rust": [
        ("scripts/qa.sh", "linecap.sh"),
        ("scripts/qa.sh", "cargo fmt --check"),
        ("scripts/qa.sh", "-D warnings"),           # clippy, warnings are errors
        ("scripts/qa.sh", "cargo check"),
        ("scripts/qa.sh", "cargo test"),
        ("scripts/e2e.sh", "cargo test"),
    ],
}


def parity_check() -> list[str]:
    """Every language profile must prove every capability in the contract."""
    problems: list[str] = []
    profiles = sorted(os.listdir(os.path.join(TEMPLATES, "profiles")))
    for lang in profiles:
        if lang not in PARITY_CONTRACT:
            problems.append(f"{lang}: profile has no PARITY_CONTRACT entry -- "
                            "add one before shipping the profile")
    for lang, rows in PARITY_CONTRACT.items():
        base = os.path.join(TEMPLATES, "profiles", lang)
        for relpath, needle in rows:
            path = os.path.join(base, relpath)
            if not os.path.isfile(path):
                problems.append(f"{lang}: parity evidence file missing: {relpath}")
                continue
            with open(path, encoding="utf-8") as f:
                content = f.read()
            if needle is not None and needle not in content:
                problems.append(f"{lang}/{relpath}: parity evidence missing: {needle!r}")
    return problems
```
